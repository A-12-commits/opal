def self.normalize(what)
  return what if Integer === what

  return  1 if what > 0
  return -1 if what < 0
  return  0
end

def ==(other)
  return true if equal?(other)

  return false unless cmp = (self <=> other)
  return Comparable.normalize(cmp) == 0
rescue StandardError
  false
end
